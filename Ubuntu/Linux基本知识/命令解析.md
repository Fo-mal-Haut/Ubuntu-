```Shell
# 第1行：将一行文本追加到 ~/.bashrc 文件末尾
echo 'export GEM_HOME="$HOME/.gem"' >> ~/.bashrc

# 分解：
# echo            - 输出文本
# 'export GEM_HOME="$HOME/.gem"'  - 要输出的文本内容
# >>             - 追加到文件（不覆盖）
# ~/.bashrc      - 用户的 bash 配置文件

# 第2行：重新加载 bash 配置
source ~/.bashrc

# 分解：
# source         - 在当前 shell 中执行文件
# ~/.bashrc      - 用户的 bash 配置文件
```
